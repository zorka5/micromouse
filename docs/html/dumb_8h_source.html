157/*
158 * DUMB provides an abstraction over files, to work with memory-mapped files,
159 * files on disk, files read into memory by other libraries in their own
160 * custom formats, ...

162 * Register your own file-handling functions as callbacks via this struct.
163 * DUMB 2.0 doesn't use long anymore. The 64-bit dumb_*_t are defined above.
164 *
165 * See DUMBFILE_SYSTEM.md in project's root for a complete spec.
166 */
167typedef struct DUMBFILE_SYSTEM {
168 /* open */
169 /* Open filename. Returned file may be of any custom type. */
170 void *(*open)(const char *filename);
171
172 /* skip */
173 /* Ignore the next n bytes in file f. Returns 0 on succes, -1 on error. */
174 int (*skip)(void *f, dumb_off_t n);
175
176 /* getc */
177 /* Read the next byte. Returns byte as unsigned, or -1 on error. */
178 int (*getc)(void *f);
179
180 /* getnc */
181 /* Read n bytes into buffer ptr. Returns number of bytes read or -1. */
182 dumb_ssize_t (*getnc)(char *ptr, size_t n, void *f);
183
184 /* close */
185 /* Called when DUMB is done with the file. User code may do anything. */
186 void (*close)(void *f);
187
188 /* seek */
189 /* Jump to offset in bytes from beginning. Returns 0 if OK, -1 on error. */
190 int (*seek)(void *f, dumb_off_t offset);

193 /* Returns the size in bytes of the file. */
194 dumb_off_t (*get_size)(void *f);
