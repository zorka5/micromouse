29/* al_ftofix and al_fixtof are used in generic C versions of al_fixmul and al_fixdiv */
30AL_INLINE(al_fixed, al_ftofix, (double x),
31{
32 if (x > 32767.0) {
33 al_set_errno(ERANGE);
34 return 0x7FFFFFFF;
35 }
36
37 if (x < -32767.0) {
38 al_set_errno(ERANGE);
39 return -0x7FFFFFFF;
40 }
41
42 return (al_fixed)(x * 65536.0 + (x < 0 ? -0.5 : 0.5));
43})

98/* In benchmarks conducted circa May 2005 we found that, in the main:
99 * - IA32 machines performed faster with one implementation;
100 * - AMD64 and G4 machines performed faster with another implementation.
101 *
102 * Benchmarks were mainly done with differing versions of gcc.
103 * Results varied with other compilers, optimisation levels, etc.
104 * so this is not optimal, though a tenable compromise.
105 *
106 * Note that the following implementation are NOT what were benchmarked.
107 * We had forgotten to put in overflow detection in those versions.
108 * If you don't need overflow detection then previous versions in the
109 * CVS tree might be worth looking at.
110 *
111 * PS. Don't move the #ifs inside the AL_INLINE; BCC doesn't like it.
112 */
113#if defined ALLEGRO_I386
114 AL_INLINE(al_fixed, al_fixmul, (al_fixed x, al_fixed y),
115 {
116 return al_ftofix(al_fixtof(x) * al_fixtof(y));
117 })
118#else
119 AL_INLINE(al_fixed, al_fixmul, (al_fixed x, al_fixed y),
120 {
121 int64_t lx = x;
122 int64_t ly = y;
123 int64_t lres = (lx*ly);
124
125 if (lres > 0x7FFFFFFF0000LL) {
126 al_set_errno(ERANGE);
127 return 0x7FFFFFFF;
128 }
129 else if (lres < -0x7FFFFFFF0000LL) {
130 al_set_errno(ERANGE);
131 return 0x80000000;
132 }
133 else {
134 return (al_fixed) (lres >> 16);
135 }
136 })
137#endif /* al_fixmul() C implementations */

140#if defined ALLEGRO_CFG_NO_FPU
141AL_INLINE(al_fixed, al_fixdiv, (al_fixed x, al_fixed y),
142{
143 int64_t lres = x;
144 if (y == 0) {
145 al_set_errno(ERANGE);
146 return (x < 0) ? -0x7FFFFFFF : 0x7FFFFFFF;
147 }
148 lres <<= 16;
149 lres /= y;
150 if (lres > 0x7FFFFFFF) {
151 al_set_errno(ERANGE);
152 return 0x7FFFFFFF;
153 }
154 else if (lres < -0x7FFFFFFF) {
155 al_set_errno(ERANGE);
156 return 0x80000000;
157 }
158 else {
159 return (al_fixed)(lres);
160 }
161})
162#else
163AL_INLINE(al_fixed, al_fixdiv, (al_fixed x, al_fixed y),
164{
165 if (y == 0) {
166 al_set_errno(ERANGE);
167 return (x < 0) ? -0x7FFFFFFF : 0x7FFFFFFF;
168 }
169 else
170 return al_ftofix(al_fixtof(x) / al_fixtof(y));
171})
172#endif
